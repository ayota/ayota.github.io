Queue Simulation

This function generates the flow of customers up to a user-specified Jth customer.

The inputs are:
  • lambda-> rate for interarrival times, which follows an exponential distribution
  • mu-> rate for service times, which follows an exponential distribution
  • N-> Desired number of simulations
  • customer-> desired number of customers in the line 
queue <- function(params = list(lambda=1, mu=.9, N=10, customer=500)) {
  queue <- data.frame()
  #loop that conducts desired number of simulations, aka N
  for (i in 1:params$N) {
    #place to hold wait times to be generated
    colnames <- c("Current Time", "Waiting Time", "Departure Time", "No. in Line", "Interarrival", "Service")
    rownames <- c(1:params$customer)
    results <- matrix(0,params$customer,6, dimnames = list(rownames,colnames))
    
    #generating exponential RVs to model interarrival times (arrivals) and service times (services)
    arrivals <-rexp(params$customer,rate=params$lambda)
    services <- rexp(params$customer,rate=params$mu)
    
    #initialize all the things
    results[1,2] <- 0
    results[1,3] <- services[1] + sum(arrivals[1]) + results[1,2]
    results[1,4] <- 1
    results[1,5] <- arrivals[1]
    results[1,6] <- services[1]
    
    #loop to generate desired number of wait times
    for (j in 2:params$customer) {
      #calculating and recording wait times
      #this formula is derived by translating the base formula, D - A, into terms of W, S, and T
      #W1 = 0
      #W2 = D1 - A2 = W1 + S1 + T1 - T1 -T2 = 0 + S1 - T2
      #W3 = D2 - A3 = W2 + S2 + T1 + T2 - T1 - T2 - T3 = W2 + S2 - T3
      #W4 = D3 - A4 = W3 + S3 + T1 + T2 + T3 -T1 -T2 -T3 - T4 = W3 + S3 - T4
      #We see all terms cancel except for W[i-1], S[i-1], and T[i]
      results[j,2] <- max(0,(results[j-1,2] + services[j-1] - arrivals[j]))
      
      #calculating the number of customers at the given time
      #departure times
      results[j,3] <- services[j] + sum(arrivals[1:j]) + results[j,2]
      
      #number of people in line
      people <- NULL
      people <- ifelse(results[,3] >= sum(arrivals[1:j]), 1, 0)
      
      results[j,4] <- sum(people)
      results[j,5] <- arrivals[j]
      results[j,6] <- services[j]
      
    }
    #calculating and recording cumulative times
    results[,1] <- cumsum(arrivals)
    
    queueTimes <- data.frame(SimNum = i, Customer=1:params$customer, Current = results[,1], Wait = results[,2], Departure= results[,3], NumLine= results[,4])
    
    queue <- rbind(queue, queueTimes)
  }
  return(queue)
}
